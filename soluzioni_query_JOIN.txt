-- 1) Selezionare tutti gli studenti iscritti al Corso di Laurea in Economia
SELECT s.*
FROM students AS s
JOIN degrees AS d ON s.degree_id = d.id
WHERE d.name = 'Corso di Laurea in Economia';

-- 2) Selezionare tutti i Corsi di Laurea Magistrale del Dipartimento di Neuroscienze
SELECT d.*
FROM degrees AS d
JOIN departments AS dp ON d.department_id = dp.id
WHERE dp.name = 'Dipartimento di Neuroscienze'
  AND LOWER(d.level) LIKE '%magistral%';


-- 3) Selezionare tutti i corsi in cui insegna Fulvio Amato (id=44)
SELECT c.*
FROM courses AS c
JOIN course_teacher AS ct ON c.id = ct.course_id
JOIN teachers AS t ON ct.teacher_id = t.id
WHERE t.id = 44;

-- 4) Selezionare tutti gli studenti con i dati relativi al corso di laurea
--    a cui sono iscritti e il relativo dipartimento,
--    in ordine alfabetico per cognome e nome
SELECT s.id AS student_id, s.name AS student_name, s.surname AS student_surname,
       d.name AS degree_name, dp.name AS department_name
FROM students AS s
JOIN degrees AS d ON s.degree_id = d.id
JOIN departments AS dp ON d.department_id = dp.id
ORDER BY s.surname, s.name;

-- 5) Selezionare tutti i corsi di laurea con i relativi corsi e insegnanti
SELECT deg.name AS degree_name,
       c.name AS course_name,
       t.name AS teacher_name, t.surname AS teacher_surname
FROM degrees AS deg
JOIN courses AS c ON c.degree_id = deg.id
JOIN course_teacher AS ct ON c.id = ct.course_id
JOIN teachers AS t ON ct.teacher_id = t.id
ORDER BY deg.name, c.name, t.surname;

-- 6) Selezionare tutti i docenti che insegnano nel Dipartimento di Matematica (54)
SELECT DISTINCT t.*
FROM teachers AS t
JOIN course_teacher AS ct ON t.id = ct.teacher_id
JOIN courses AS c ON ct.course_id = c.id
JOIN degrees AS d ON c.degree_id = d.id
JOIN departments AS dp ON d.department_id = dp.id
WHERE dp.id = 54;

-- 7) BONUS: Per ogni studente il numero di tentativi sostenuti per ogni esame,
--           stampando anche il voto massimo.
--           Poi filtrare i tentativi con voto minimo 18.
SELECT s.id AS student_id,
       s.name AS student_name,
       s.surname AS student_surname,
       c.name AS course_name,
       COUNT(er.id) AS num_attempts,
       MAX(er.grade) AS max_grade
FROM students AS s
JOIN exam_results AS er ON s.id = er.student_id
JOIN exam_sessions AS es ON er.exam_session_id = es.id
JOIN courses AS c ON es.course_id = c.id
GROUP BY s.id, s.name, s.surname, c.name
HAVING MIN(er.grade) >= 18
ORDER BY s.surname, s.name, c.name;
